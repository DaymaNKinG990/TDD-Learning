import requests
import time

def wait_for_service(url: str, timeout: int = 300) -> bool:
    """–û–∂–∏–¥–∞–Ω–∏–µ –≥–æ—Ç–æ–≤–Ω–æ—Å—Ç–∏ —Å–µ—Ä–≤–∏—Å–∞."""

    while time.time() - start_time < timeout:

            response = requests.get(url)
            if response.status_code == 200:
                return True
        except requests.RequestException:
            pass

        time.sleep(10)

    return False
